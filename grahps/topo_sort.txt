class Solution
{
	public:
	//Function to return list containing vertices in Topological order.
	void dfs(int v,vector<int> adj[],int vis[],stack<int> &st)
	{
	     vis[v]=1;
	     
    	  for(auto i: adj[v])
    	     {
    	        if(vis[i]==0)
	            {
            	      dfs(i,adj,vis,st);
	            }
    	     }
    	 st.push(v);
	}
	vector<int> topoSort(int v, vector<int> adj[]) 
	{
	    // code here
	    vector<int> ans;
	    int vis[v]={0};
	    stack<int> st;
	    for(int i=0;i<v;i++)
	    {
	        if(vis[i]==0)
	         {
    	        dfs(i,adj,vis,st);
	         }
	    }
	    
	   while(!st.empty())
	   {
	       ans.push_back(st.top());
	       st.pop();
	   }
	    return(ans);
	}
};
