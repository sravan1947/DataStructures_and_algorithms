class Solution{

	int NthRoot(int n, int m)
	{
	    // Code here.
	    int i=0,j=m;
	    int c=10;
	    while(i<=j)
	    {
	        int mid=((i+j)/2);
	       // cout<<mid<<" "<<n<<" "<<pow(mid,n)<<endl;
	        if(pow(mid,n)==m)
	            return(mid);
	       
	        else
	        if(pow(mid,n)<m)
	        {
	            i=mid+1;
	        }
	        else
	        if(pow(mid,n)>m)
	        {
	            j=mid-1;
	        }
	        
	    }
	    return(-1);
	}  
};
